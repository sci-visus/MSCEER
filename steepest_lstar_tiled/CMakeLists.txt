CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET(CMAKE_CXX_STANDARD 20)

ENABLE_LANGUAGE(CXX)

if(APPLE)
  SET(OpenMP_C_FLAGS "")
  SET(OpenMP_CXX_FLAGS "")
  SET(OpenMP_EXE_LINKER_FLAGS "-fopenmp")
else()
  FIND_PACKAGE( OpenMP REQUIRED )
endif()

SET(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${OpenMP_C_FLAGS}")
SET(CMAKE_CXX_FLAGS "--std=c++20 ${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")
SET(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} ${OpenMP_EXE_LINKER_FLAGS}")

FILE(GLOB INCLUDES ${GInt_SOURCE_DIR}/include/*.h)

SET(HEADERS ${INCLUDES})
INCLUDE_DIRECTORIES(${GInt_SOURCE_DIR}/include/)

FILE(GLOB SOURCES ${GInt_SOURCE_DIR}/steepest_lstar_tiled/*.cxx)

ADD_EXECUTABLE(Steepest_lstar_tiled ${HEADERS} ${SOURCES})
  
TARGET_LINK_LIBRARIES(Steepest_lstar_tiled LINK_PUBLIC GInt)
